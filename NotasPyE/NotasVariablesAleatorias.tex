\begin{Ejer}
Calcular $\mathbb{E}[X]$  para $f_X(x)=C\,e^{-x^2},\; x\in\mathbb{R}$.

Primero hallamos la constante de normalización \(C\), lo cual requiere conocer
\(\displaystyle \int_{-\infty}^{\infty} e^{-x^2}\,dx\).

A saber  \(\displaystyle \int_{-\infty}^{\infty} e^{-x^2}\,dx=\sqrt{\pi}\).

Sea
\[\int_{-\infty}^{\infty} e^{-x^2}\,dx.\]
Como el integrando es no negativo, por el teorema de Tonelli/Fubini podemos escribir
\[
I^2 = \left(\int_{-\infty}^{\infty} e^{-x^2}\,dx\right)\left(\int_{-\infty}^{\infty} e^{-y^2}\,dy\right)=\int_{-\infty}^{\infty}\int_{-\infty}^{\infty} e^{-(x^2+y^2)}\,dx\,dy.
\]
Interpretamos la integral doble en el plano \(\mathbb{R}^2\) y pasamos a coordenadas polares:
\[
x = r\cos\theta,\qquad y = r\sin\theta,\qquad r\in[0,\infty),\ \theta\in[0,2\pi).
\]

El jacobiano de la transformación \((r,\theta)\mapsto(x,y)\) es
\begin{eqnarray}
J(r,\theta)
=det\left(\begin{array}{cc}
\frac{\partial x}{\partial r} & \frac{\partial x}{\partial \theta}\\
\frac{\partial y}{\partial r} & \frac{\partial y}{\partial \theta}
\end{array}
\right)
=\left(\begin{array}{cc}
\cos\theta & -r\sin\theta\\
\sin\theta & \phantom{-}r\cos\theta
\end{array}\right)
= r(\cos^2\theta+\sin^2\theta)=r.
\end{eqnarray}
Por tanto, \(dx dy = \vert J\vert dr d\theta = r dr d\theta\).

Además, \(x^2+y^2=r^2\), de modo que
\[
I^2
= \int_{0}^{2\pi}\int_{0}^{\infty} e^{-r^2}\,r\,dr\,d\theta
= \left(\int_{0}^{2\pi} d\theta\right)\left(\int_{0}^{\infty} r e^{-r^2} dr\right)
= 2\pi \int_{0}^{\infty} r e^{-r^2} dr.
\]
Hacemos el cambio \(u=r^2\) (de modo que \(du=2r dr\) y \(rdr=\frac{1}{2}du\)):
\[
\int_{0}^{\infty} r e^{-r^2} dr
= \frac{1}{2}\int_{0}^{\infty} e^{-u} du
= \frac{1}{2}[-e^{-u}]_{0}^{\infty}
= \frac{1}{2}.
\]
Así,
\[
I^2 = 2\pi\cdot\frac12 = \pi\Longrightarrow I=\int_{-\infty}^{\infty} e^{-x^2}dx=\sqrt{\pi}.
\]


Imponiendo que \(f_X\) integre 1 en \(\mathbb{R}\):
\[
1=\int_{-\infty}^{\infty} C e^{-x^2} dx = C\int_{-\infty}^{\infty} e^{-x^2} dx
= C \sqrt{\pi} \Rightarrow C=\frac{1}{\sqrt{\pi}}.
\]
Por tanto,
\begin{eqnarray*}
f_X(x)=\frac{1}{\sqrt{\pi}} e^{-x^2}\textrm{ , con }x\in\mathbb{R}.
\end{eqnarray*}

La esperanza  \(\mathbb{E}[X]\),  por definición,
\[
\mathbb{E}[X]
= \int_{-\infty}^{\infty} x\,f_X(x)\,dx
= \frac{1}{\sqrt{\pi}}\int_{-\infty}^{\infty} x\,e^{-x^2}\,dx.
\]
Observamos que \(x\,e^{-x^2}\) es una función \emph{impar}:
\[
x\,e^{-x^2} = -(-x)\,e^{-(-x)^2}.
\]
La integral de una función impar en un intervalo simétrico es cero, así que
\[
\int_{-\infty}^{\infty} x\,e^{-x^2}\,dx = 0
\quad\Longrightarrow\quad
\mathbb{E}[X]=0.
\]

\end{Ejer}


\subsection*{Distribuci\'on Bernoulli}

\begin{Def} Sea $X$ variable alatoria con distribuci\'on Bernoulli$(p)$ $x \in \{0,1\}$,  entonces $p_X(x)=p^{x}(1-p)^{1-x}$,  $F_X(x) = \left\{
\begin{array}{cc}
0, & x<0 \\
1-p, & 0\leq x < 1 \\
1, & x \geq 1
\end{array}
\right. $ entonces $\mathbb{E}[X]=p$ y $Var(X) = p(1-p)$; adem\'as $M_X(t) = \mathbb{E}[e^{tX}] = (1-p) + pe^{t} $ y  $G_X(s) = \mathbb{E}[s^X] = (1-p) + ps$.
\end{Def}

\subsection*{Distribuci\'on Binomial}

\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye Binomial$(n,p)$, $x \in \{0,1,\dots,n\} $,  entonces $p_X(x) =\binom{n}{x}p^{x}(1-p)^{n-x}$ y $F_X(x)=\sum_{k=0}^{\lfloor x \rfloor}\binom{n}{k} p^{k}(1-p)^{n-k}$.  Entonces  $\mathbb{E}[X]=np$; $\mathrm{Var}(X)=np(1-p)$, adem\'as $M_X(t) = \left((1-p)+pe^{t}\right)^{n}$ y $G_X(s) = \left((1-p)+ps\right)^{n}$.
\end{Def}


\subsection*{Distribuci\'on Geom\'etrica}
\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye Geom\'etrica$(p)$;  $x \in \{1,2,3,\dots\}$, entonces
$p_X(x)=p(1-p)^{x-1}$, $F_X(x)=1-(1-p)^{\lfloor x \rfloor}$. Entonces $\mathbb{E}[X]=\frac{1}{p}$, $\mathrm{Var}(X)= \frac{1-p}{p^{2}}$,  adem\'as $M_X(t) = \frac{pe^{t}}{1-(1-p)e^{t}}$ ,para $ e^{t}<\frac{1}{1-p}$ y 
$G_X(s)=\frac{ps}{1-(1-p)s}$, para  $|s|<\frac{1}{1-p}$.
\end{Def}


\subsection*{Distribuci\'on Poisson}

\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye Poisson$(\lambda)$, $x \in \{0,1,2,\dots\}$, entonces $p_X(x)=e^{-\lambda}\frac{\lambda^{x}}{x!}$,  $F_X(x) =\sum_{k=0}^{\lfloor x \rfloor} e^{-\lambda}\frac{\lambda^{k}}{k!} $, entonces $\mathbb{E}[X]= \lambda$,  $\mathrm{Var}(X) = \lambda$, adem\'as $M_X(t) = \exp\!\left(\lambda(e^{t}-1)\right)$ y $G_X(s) = \exp\!\left(\lambda(s-1)\right)$.
\end{Def}

\subsection*{Distribuci\'on Binomial Negativa}

\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye Binomial Negativa$(r,p)$; fracasos hasta el $r$-\'esimo \'exito, para $x \in \{0,1,2,\dots\}$, entonces 
$ p_X(x) = \binom{x+r-1}{x}(1-p)^{x} p^{r}$, $F_X(x)= \sum_{k=0}^{\lfloor x \rfloor}\binom{k+r-1}{k}(1-p)^{k} p^{r}$, adem\'as $\mathbb{E}[X]=r\frac{1-p}{p}$, $\mathrm{Var}(X) = r\frac{1-p}{p^{2}}$, adem\' as $M_X(t)= \left(\frac{p}{1-(1-p)e^{t}}\right)^{r}$,  para $ e^{t}<\frac{1}{1-p}$ y $G_X(s) = \left(\frac{p}{1-(1-p)s}\right)^{r}$, para $|s|<\frac{1}{1-p}$.
\end{Def}

\subsection*{Distribuci\'on Uniforme}

\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye Uniforme$(a,b)$, $x \in [a,b]$,  entonces $f_X(x)= \frac{1}{b-a}$, $F_X(x)=\left\{
\begin{array}{cc}
0, & x<a \\
\frac{x-a}{b-a}, & a \le x < b \\
1, & x \ge b
\end{array}
\right. $, entonces $\mathbb{E}[X] = \frac{a+b}{2}$ y $\mathrm{Var}(X) = \frac{(b-a)^{2}}{12}$; adem\'as $M_X(t) = \frac{e^{tb}-e^{ta}}{t(b-a)}$,  para $t \neq 0$  y  $M_X(0)=1$,  la  $PGF$ no existe.
\end{Def}

\subsection*{Distribuci\'on Exponencial}

\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye Exponencial$(\lambda)$, $x \ge 0$, entonces $f_X(x)= \lambda e^{-\lambda x}$ y $F_X(x) = 1-e^{-\lambda x}$, entonces $\mathbb{E}[X] =\frac{1}{\lambda}$, y  $\mathrm{Var}(X) =\frac{1}{\lambda^{2}}$, adem\'as $M_X(t) = \frac{\lambda}{\lambda - t}$,  para $t<\lambda$,  la $PGF$ no se puede calcular.
\end{Def}


\subsection*{Distribuci\'on Normal}
\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye Normal$(\mu,\sigma^{2})$,  $x \in \mathbb{R}$
$f_X(x) =\frac{1}{\sqrt{2\pi}\sigma}\exp\!\left(-\frac{(x-\mu)^{2}}{2\sigma^{2}}\right)$, $F_X(x) =\Phi\!\left(\frac{x-\mu}{\sigma}\right)$ (funci\'on de distribuci\'on normal est\'andar), entonces $\mathbb{E}[X] =\mu$ y $\mathrm{Var}(X) = \sigma^{2}$; adem\'as $M_X(t) = \exp\!\left(\mu t + \frac{\sigma^{2}t^{2}}{2}\right)$,  la $PGF$ no se puede calcular.
\end{Def}

\subsection*{Distribuci\'on Gamma} 
\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye Gamma$(k,\theta)$; (forma $k>0$, escala $\theta>0$), $x>0$
$f_X(x) = \frac{1}{\Gamma(k)\theta^{k}} x^{k-1} e^{-x/\theta}$, y $F_X(x) = \frac{\gamma\!\left(k, x/\theta\right)}{\Gamma(k)}$(gamma incompleta regularizada), entonces $\mathbb{E}[X] = k\theta$,  $ \mathrm{Var}(X) = k\theta^{2}$, adem\'as $M_X(t) = (1-\theta t)^{-k}$ , para $t<\frac{1}{\theta}$,  la $PGF$ no se puede calcular.
\end{Def}

\subsection*{Distribuci\'on $\chi^{2}$}
\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye $\chi^{2}(\nu)$, $x > 0$, entonces
$f_X(x) = \frac{1}{2^{\nu/2}\Gamma(\nu/2)} x^{\nu/2 - 1} e^{-x/2}$ y $F_X(x) = \frac{\gamma\!\left(\nu/2, x/2\right)}{\Gamma(\nu/2)}$; adem\'as $\mathbb{E}[X] = \nu$,  $\mathrm{Var}(X) = 2\nu$ y $M_X(t) = (1-2t)^{-\nu/2}$,para $t<\frac{1}{2}$,  la $PGF$ no se puede calcular.
\end{Def}


\subsection*{Distribuci\'on Beta}
\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye Beta$(\alpha,\beta)$,  $x \in (0,1)$
$f_X(x) = \frac{1}{B(\alpha,\beta)} x^{\alpha-1}(1-x)^{\beta-1}$, $F_X(x) = I_{x}(\alpha,\beta)$ (beta incompleta regularizada), entonces $\mathbb{E}[X] = \frac{\alpha}{\alpha+\beta}$ y $\mathrm{Var}(X) = \frac{\alpha\beta}{(\alpha+\beta)^{2}(\alpha+\beta+1)}$; adem\'as $M_X(t) = {}_{1}F_{1}(\alpha;\,\alpha+\beta;\,t)$,  la $PGF$ no se puede calcular.
\end{Def}

\subsection*{Distribuci\'on Hipergeom\'etrica}
\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye Hipergeom\'etrica$(N,M,n)$,  Par\'ametros: $N$ total,  $M$ \'exitos en la poblaci\'on, $n$ extracciones sin reemplazo.   $x \in \{\max(0,\,n-(N-M)),\,\dots,\,\min(n,\,M)\}$.  $p_X(x) = \binom{M}{x}\binom{N-M}{n-x}\binom{N}{n}$ y 

$F_X(x) = \sum_{k=\max(0,\,n-(N-M))}^{\lfloor x \rfloor}\binom{M}{k}\binom{N-M}{n-k}\binom{N}{n}$, entonces  $\mathbb{E}[X] = n\frac{M}{N}$ y 

$\mathrm{Var}(X) = n\frac{M}{N}\!\left(1-\frac{M}{N}\right)\!\frac{N-n}{N-1}$; adem\'as $M_X(t) = \sum_{x} e^{tx}\frac{\binom{M}{x}\binom{N-M}{n-x}}{\binom{N}{n}}$; adem\'as $G_X(s) = \sum_{x} s^{x}
\frac{\binom{M}{x}\binom{N-m}{n-x}}{\binom{N}{n}}$

\end{Def}

\subsection*{Distribuci\'on Weibull}

\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye  Weibull$(k,\lambda)$; (forma $k>0$, escala $\lambda>0$), $x\geq0$
$f_X(x) = \frac{k}{\lambda}\left(\frac{x}{\lambda}\right)^{k-1}\exp\!\left[-\left(\frac{x}{\lambda}\right)^{k}\right]$,  $F_X(x) = 1-\exp\!\left[-\left(\frac{x}{\lambda}\right)^{k}\right]$, entonces $\mathbb{E}[X] = \lambda\,\Gamma\!\left(1+\frac{1}{k}\right)$ y $\mathrm{Var}(X) = \lambda^{2}\!\left[\Gamma\!\left(1+\frac{2}{k}\right)-\Gamma\!\left(1+\frac{1}{k}\right)^{2}\right]$, adem\'as
$M_X(t) = \sum_{n=0}^{\infty} \frac{t^{n}}{n!}\,\lambda^{n}\,\Gamma\!\left(1+\frac{n}{k}\right)$,  la PGF no se puede calcular
\end{Def}

\subsection*{Distribuci\'on Log-normal}
\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye Lognormal$(\mu,\sigma^{2})$, $x>0$
$f_X(x) = \frac{1}{x\,\sigma\sqrt{2\pi}} \exp\!\left[-\frac{(\ln x-\mu)^{2}}{2\sigma^{2}}\right]$ y $F_X(x) = \Phi\!\left(\frac{\ln x-\mu}{\sigma}\right)$; entonces $\mathbb{E}[X] = \exp\!\left(\mu+\frac{\sigma^{2}}{2}\right)$ y $\mathrm{Var}(X) =\left(e^{\sigma^{2}}-1\right)e^{2\mu+\sigma^{2}}$; adem\'as $M_X(t) = \mathbb{E}[e^{tX}]$; no existe para $t>0$,  la PGF no se puede calcular.

\end{Def}

\subsection*{Distribuci\'on $t$ de Student}

\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye $t$ de Student$(\nu)$, $x \in \mathbb{R}$
$f_X(x) = \frac{\Gamma\!\left(\frac{\nu+1}{2}\right)}{\sqrt{\nu\pi}\,\Gamma\!\left(\frac{\nu}{2}\right)} \left(1+\frac{x^{2}}{\nu}\right)^{-\frac{\nu+1}{2}}$ y $F_X(x) = T_{\nu}(x)$ (CDF $t$ con  $\nu$ g.l.),  entonces $ \mathbb{E}[X] = 0$ si  $\nu>1$ (indefinida si  $\nu \leq 1$); y $\mathrm{Var}(X) = \frac{\nu}{\nu-2}$ si $\nu>2$; $\infty$ si $1<\nu\leq 2$; indefinida si $\nu\leq 1$, adem\'as $M_X(t)$ no existe y la  PGF no se puede calcular.

\end{Def}

\subsection*{Distribuci\'on Cauchy}
\begin{Def} Sea $X$ variable aleatoria, tal que se distribuye Cauchy$(x_{0},\gamma)$; locaci\'on $ x_{0}\in\mathbb{R}$,  escala $\gamma>0$, $x \in \mathbb{R}$, $ f_X(x) = \frac{1}{\pi \gamma}\,\frac{1}{1+\left(\frac{x-x_{0}}{\gamma}\right)^{2}}$ y $F_X(x) = \frac{1}{\pi}\arctan\!\left(\frac{x-x_{0}}{\gamma}\right) + \frac{1}{2}$, adem\'as $\mathbb{E}[X]$ no est\'a definida,  $\mathrm{Var}(X)$ No est\'a definida,  $M_X(t)$ no est\'a definida y la PGF no se puede calcular.

\end{Def}

\subsubsection{Distribución Uniforme}
%<>==<><>==<><>==<><>==<><>==<><>==<><>==<>

\begin{Ejer} Resuelve los siguientes problemas relacionados con la variable aleatoria uniforme discreta.

\begin{enumerate}
    \item Un juego de dados justo se lanza tres veces. ¿Cuál es la probabilidad de obtener una suma de 10?
    \item Una urna contiene 10 bolas numeradas del 1 al 10. Si se elige una bola al azar, ¿cuál es la probabilidad de que sea un número par?
    \item Un fabricante de piezas de repuesto sabe que el 5\% de las piezas que produce son defectuosas. Si se eligen aleatoriamente 5 piezas, ¿cuál es la probabilidad de que al menos una de ellas sea defectuosa?
    \item Un fabricante de cartas de póker quiere asegurarse de que todas las cartas tengan la misma probabilidad de ser seleccionadas al azar. ¿Cuántas cartas debe incluir en su mazo?
    \item Un estudiante elige al azar una página de un libro y cuenta el número de palabras que contiene. Si el libro tiene 200 páginas y un promedio de 1000 palabras por página, ¿cuál es la probabilidad de que el número de palabras en una página elegida al azar esté entre 900 y 1100?
    \item Un vendedor de seguros tiene 10 pólizas que puede vender. Si se eligen al azar 3 pólizas, ¿cuál es la probabilidad de que al menos una de ellas sea vendida?
    \item Un fabricante de dulces quiere que cada bolsa contenga una cantidad igual de caramelos de cada sabor. Si hay 4 sabores de caramelos y cada bolsa contiene 20 caramelos, ¿cuál es la probabilidad de que haya exactamente 5 caramelos de cada sabor en una bolsa seleccionada al azar?
